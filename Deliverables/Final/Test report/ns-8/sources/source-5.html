496         @Override

498             if(o1.getDiscardPileHead() == null && o2.getDiscardPileHead() != null)
499                 return 0;
500             if(o1.getDiscardPileHead() == null && o2.getDiscardPileHead() != null)

505                 return 0;
506             if(!o1.isFromActualTurn() && o2.isFromActualTurn())
507                 return 1;

509                 return -1;

511                 for(int i = game.getFirstPlayerInPlanning(); i != Math.floorMod(game.getFirstPlayerInPlanning() - 1, game.getPlayersCount()); i = Math.floorMod(i + 1, game.getPlayersCount())){
512                     if(game.getPlayer(i).getNickname().equals(o1.getNickname()))
513                         return -1;
514                     if(game.getPlayer(i).getNickname().equals(o2.getNickname()))
515                         return 1;
516                 }
517             }
518             return o1.getDiscardPileHead().value() - o2.getDiscardPileHead().value();
519         }

646     /**
647      * If a player has fewer towers than the other players, wins
648      * If two players have the same number of towers, but one has more professors, wins
649      * If two players have the same number of towers and professors, they tied
650      * @return the nickname of the winner/winners
651      */
652     private List<String> whoIsWinning() {
653         List<String> winners = new ArrayList<>();
654 
655         //check number of tower left
656         List<Player> playerSortedByTower = Arrays.stream(game.getPlayers()).
657                 sorted(Comparator.comparingInt(p -> p.getBoard().getTowersCount())).toList();
658         if (playerSortedByTower.get(0).getBoard().getTowersCount() < playerSortedByTower.get(1).getBoard().getTowersCount()) {
659             winners.add(playerSortedByTower.get(0).getNickname());
660             return winners;
661         }
662 
663         List<Player> playerSortedByProfessors;
664         if(game.getParameters().getPlayersNumber().getPlayersNumber() == 3
665                 && playerSortedByTower.get(1).getBoard().getTowersCount() == playerSortedByTower.get(2).getBoard().getTowersCount())
666             playerSortedByProfessors = playerSortedByTower.subList(0, 3);
667         else
668             playerSortedByProfessors = playerSortedByTower.subList(0, 2);
669         playerSortedByProfessors.stream().
670                 sorted(Comparator.comparing(Player::getProfessorsCount).reversed()).toList();
671 
672         if (playerSortedByProfessors.get(0).getBoard().getProfessorsCount() < playerSortedByProfessors.get(1).getBoard().getProfessorsCount()) {
673             winners.add(playerSortedByTower.get(1).getNickname());
674             return winners;
675         }
676 
677 
678         // it arrives here only if there are 2 or 3 player with the same number of tower and professors
679         winners.add(playerSortedByTower.get(0).getNickname());
680         winners.add(playerSortedByTower.get(1).getNickname());
681         if(game.getParameters().getPlayersNumber().getPlayersNumber() == 3
682                 && playerSortedByProfessors.get(1).getBoard().getProfessorsCount() == playerSortedByProfessors.get(2).getBoard().getProfessorsCount())
683             winners.add(playerSortedByTower.get(2).getNickname());
684         return winners;
